\newcommand{\abilityscore}[3]{

      ({\secondfold + \foldwidth/8 * 2.5}, {#1 - \textheight/14}) --
      ({\secondfold + \foldwidth/8 * 2.25}, {#1 - \textheight/14 + \textheight/40}) --

\node[align=right,left] at ({\secondfold + \foldwidth}, {#1 - \textheight/28}) {#3};
